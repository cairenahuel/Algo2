\aux{unidadFondoComun}{recursos: \TLista{\float}, cooperan: \TLista{\bool}}{\float}{\\(\sum_{i=0}^{|recursos|-1} \IfThenElse{cooperan[i]}{recursos[i]}{0})/|recursos|}
\hfill\\
\pred{enRango}{i: \ent, lista: \TLista{\nat}}{0\leq i<|lista|}
\hfill\\
\pred{todosPositivos}{lista: \TLista{\float}}{
\paraTodo[unalinea]{i}{\ent}{enRango(i, lista)\yLuego lista[i]>0}
}
\pred{esRectangular}{matriz: \TLista{\TLista{T}}}{\paraTodo[unalinea]{i}{\ent}{enRango(i,tail(matriz))\yLuego |matriz[i]|=|matriz[i+1]|}}
\hfill\\
\aux{cuantoGanaron}{recursosPaso: \TLista{\float}, apuestas: \TLista{\TLista{\float}}, pagos: \TLista{\TLista{\float}}, eventosPaso:  \TLista{\nat}}{\TLista{\float}}{
\\\hspace*{2em}\sum_{i=0}^{|recursosPaso|-1}{\langle recursosPaso[i]\cdot apuestas[i][eventosPaso[i]] \cdot  pagos[i][eventosPaso[i]] \rangle}
}

\aux{ufcPaso}{coop: \TLista{\bool}, recPaso: \TLista{\float}, apuestas: \TLista{\TLista{\float}}, pagos: \TLista{\TLista{\float}},evtPaso :  \TLista{\nat} }{\float}{

\pred{esApuestaValida}{\In apuesta \TLista{\float}}{
	(\paraTodo[unalinea]{i}{\nat}{enRango(i, apuesta) \yLuego apuesta[i] \geq 0}) \yLuego (\sum_{i=0}^{|apuesta|-1}{apuesta[i]} = 1)
}
\hfill\\
\aux{distribucionPaso}{cosas}{\TLista{\float}}{\\\sum_{i=0}^{|coop|-1}{\langle  \IfThenElse{coop[i]}{ufcPaso(cosas)}{cuantoGanaron(cosas)[i]+ufcPaso(cosas)}\rangle}}
\hfill\\
\aux{trayectoriaFinal}{cosas}{\TLista{\TLista{\float}}}{\sum_{i=0}^{|eventos|-1}\langle distribucionPaso(cosas)\rangle}
\hfill\\
\aux{cuantoGanoAlFinal}{\In trayectorias: \TLista{\TLista{\float}}, \In individuo: \nat}{\float}{
	\\\hspace*{2em}(trayectorias[i][|trayectorias[0]|-1) - trayectorias[i][0]]
}
\hfill\\
\aux{distribucionPaso}{in p, in coop: \TLista{\bool},in recPaso: \TLista{\float},in apuestas: \TLista{\TLista{\float}},in pagos: \TLista{\TLista{\float}},in evtPaso :  \TLista{\nat}}{\TLista{\float}}{\sum_{i=0}^{|coop|-1} \langle \IfThenElse{coop[i]\\\hspace*{2em}}{ufcPaso(coop, recPaso, apuestas, pagos, evtPaso)\\\hspace*{2em}}{cuantoGanaron(recPaso, apuestas, pagos, evtPaso)[i]+ufcPaso(coop, recPaso, apuestas, pagos, evtPaso)} \rangle}
\hfill\\

\begin{proc}{redistribucionDeLosFrutos}{\In recursos : \TLista{\float}, \In cooperan: \TLista{\bool}}{\TLista{\float}}
\requiere{|recursos|=|cooperan|}
\requiere{|recursos|>0}
\requiere{todosPositivos(recursos)}
\asegura{|res|=|recursos|}
\asegura{\paraTodo[unalinea]{\ individuo}{\ent}{enRango(individuo, |cooperan|) \implicaLuego res[individuo]=
\\\bigl(unidadFondoComun(recursos, cooperan)+\IfThenElse{cooperan[individuo]}{0}{recursos[n]}\bigr)}}
\salto{\baselineskip}
\aux{unidadFondoComun}{recursos: \TLista{\float}, cooperan: \TLista{\bool}}{\float}{
    \\(\sum_{i=0}^{|recursos|-1} \IfThenElse{cooperan[i]}{recursos[i]}{0})/|recursos|}
\end{proc}

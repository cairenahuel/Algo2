\requiere{|trayectoria|>0\land todosPositivos(trayectoria)}
\asegura{
res=\True \iff 
\existe[unalinea]{n}{\ent}{esMaximoLocal(trayectoria,n) \land

\pred{existeAnterior}{\In s:\TLista{\float}, \In n:\nat}
{
enRango(n,s) \yLuego enRango(n-1,s)
}
\pred{existePosterior}{\In s:\TLista{\float}, \In n:\nat}
{
enRango(n,s) \yLuego  enRango(n+1,s)
}
\pred{esMaximoLocal}{\In s:\TLista{\float}, \In n:\nat}{
    (existeAnterior(s,n) \implies s[n-1] < s[n])
    \yLuego
    (existePosterior(s,n) \implies s[n+1] < s[n])
}

\begin{proc}{trayectoriaExtraÃ±aEscalera}{\In trayectoria: \TLista{\float}}{\bool}
\requiere{|trayectoria|>0\land todosPositivos(trayectoria)}
\asegura{
res=\True \iff 
\existe[unalinea]{\ n}{\ent}{esMaximoLocal(trayectoria,n) \land
\\\hspace*{12em}\paraTodo[unalinea]{\ m}{\ent}{m\not=n\land \lnot esMaximoLocal(trayectoria,m)}}
}
\salto{\baselineskip}
\pred{existeAnterior}{secuencia: \TLista{\float}, n: \nat}
{
enRango(n,|secuencia|) \yLuego enRango(n-1,|secuencia|)
}
\pred{existePosterior}{secuencia: \TLista{\float}, n: \nat}
{
enRango(n,|secuencia|) \yLuego  enRango(n+1,|secuencia|)
}
\pred{esMaximoLocal}{secuencia: \TLista{\float}, n: \nat}{
    (existeAnterior(secuencia,n) \implies secuencia[n-1] < secuencia[n])
    \yLuego
    \\(existePosterior(secuencia,n) \implies secuencia[n+1] < secuencia[n])
}
